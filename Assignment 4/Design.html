import java.awt.Color;
import java.awt.Graphics;
import javax.swing.JPanel;
import javax.swing.JFrame;

class LinesRectsOvalsJPanel extends JPanel 
{
   
   @Override
   public void paintComponent(Graphics g)
   {
     super.paintComponent(g); 
      this.setBackground(Color.PINK);
	
	g.setColor(Color.GREEN);
        g.fillRect(30,30,70,50);
	g.setColor(Color.RED);
	g.fillRect(80,30,50,100);
	g.setColor(Color.GREEN);
	g.fillRect(80,120,140,50);
	g.setColor(Color.YELLOW);
	g.fillOval(130,30,50,90);
	g.setColor(Color.RED);
	g.fillRect(180,120,50,100);
	g.setColor(Color.GREEN);
	g.fillRect(180,200,140,50);
	g.setColor(Color.YELLOW);
	g.fillOval(230,120,50,90);
	
	

	
	
     
}
}
public class Design
{
   
   public static void main(String[] args)
   {
      JFrame frame = 
         new JFrame("Design");
      frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
   
      LinesRectsOvalsJPanel lROJPanel = 
         new LinesRectsOvalsJPanel(); 
      lROJPanel.setBackground(Color.WHITE); 
      frame.add(lROJPanel); 
      frame.setSize(500, 500);
      frame.setVisible(true);
   } 
}
